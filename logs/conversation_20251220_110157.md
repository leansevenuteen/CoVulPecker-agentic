    strcpy(ptr, "Hello World");
    free(ptr);
    printf("%s\n", ptr);  // Use after free!

void integer_overflow(int size) {
    if (size < 0) return;

    int buffer_size = size + 100;

    if (buffer == NULL) return;
    
    // This could write beyond allocated memory
    memset(buffer, 'A', size);
    free(buffer);
}
